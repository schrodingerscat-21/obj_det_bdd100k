# Batch processing
python yolo_inference.py --model "model.pt" --source "test_images/" --output "results/" --pattern "*.jpg"

python yolo_inference.py --model "model.pt" --source "test_video.mp4" --output "results/" --show

# Custom thresholds
python yolo_inference.py --model "model.pt" --source "image.jpg" --conf 0.3 --iou 0.5 --device "cpu"


programmatic usage:

from yolo_inference import BDD100KInference

# Initialize
inference = BDD100KInference(
    model_path="path/to/your/trained_model.pt",
    device="0",
    conf_threshold=0.25,
    iou_threshold=0.7
)

# Process single image
result = inference.predict_image("test.jpg", save_path="output.jpg", show=True)

# Batch process
results = inference.predict_batch("images/", "outputs/", "*.jpg")
